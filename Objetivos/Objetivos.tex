\chapter*{Objetivos}
%mirar problema con fancyheader y secciones no numeradas, pagina 21-22 del manual
\fancyhead[RO,LE]{\sc{Objetivos}}

\addcontentsline{toc}{chapter}{Objetivos}
%\cabeceraEspecial{Objetivos}


\begin{itemize}
  \item Desarrollo de una aplicación web, denominada Proteopathogen (v 1.0),
  respaldada por una base de datos
  para recoger, almacenar y analizar resultados de identificación de 
  péptidos y proteínas procedentes de experimentos de proteómica
  relacionados con el estudio de la interacción hospedador-patógeno usando
  el modelo de \mbox{\textit{Candida albicans}}
  \item Adopción del formato estándar de identificaciones de péptidos y proteínas mzIdentML
  como fuente de datos para la base de datos y aplicación web Proteopathogen (v 2.0)
  \item Recopilación de resultados de espectrometría de masas y  %no resultados de identificaciones, sino los espectros
  creación de un \mbox{PeptideAtlas}, o Atlas Peptidico, de \mbox{\textit{Candida albicans}}
  \item Reanálisis de los datos existentes en el PeptideAtlas original
  usando una nueva base de datos con secuencias específicas de alelo, un
  procesamiento multi-algoritmo y resultados de nuevos experimentos diseñados \textit{ad hoc}
  para incrementar la cobertura del proteoma.
\end{itemize}
